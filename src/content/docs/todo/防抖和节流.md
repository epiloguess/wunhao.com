#### 防抖

```js
function debounce(callback, delay = 1000) {
  var timerid;
  return (...args) => {
    clearTimeout(time);
    timerid = setTimeout(() => {
      callback(...args);
    }, delay);
  };
}
```

#### 节流

```js
function throttle(callback, delay = 1000) {
  let shouldWait = false;

  return (...args) => {
    if (shouldWait) return;
    callback(...args);
    shouldWait = true;
    setTimeout(() => {
      shouldWait = false;
    }, delay);
  };
}
```
